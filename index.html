    function parseLatLng(baseData) {

      let lat = NaN, lng = NaN;

      if (!baseData) return { lat, lng };

      const latRaw = baseData["緯度"];

      const lngRaw = baseData["經度"];

      if (latRaw !== "" && lngRaw !== "" && latRaw != null && lngRaw != null) {

        lat = parseFloat(latRaw);

        lng = parseFloat(lngRaw);

      } else if (baseData["GPS"]) {

        const parts = baseData["GPS"].toString().split(",");

        lat = parseFloat((parts[0] || "").trim());

        lng = parseFloat((parts[1] || "").trim());

      }

      return { lat, lng };

    }

    function renderMarkers(bases) {

      const markerGroup = L.featureGroup();

      const pinHtml = (w, h, fill, showQuest) => {

        const questTag = showQuest ? ` 

          <g transform="translate(14, 0)"> 

            <circle cx="5" cy="5" r="5" fill="#e2715d" stroke="#FFFFFF" stroke-width="1"/>

          </g> ` : "";

        return ` 

          <div style="width:${w}px; height:${h}px; display:flex; justify-content:center; align-items:center;"> 

            <svg viewBox="0 0 24 24" width="${w}" height="${h}" style="display:block; overflow:visible;">

              <circle cx="12" cy="9" r="3" fill="#FFFFFF"/> 

              ${questTag} 

            </svg> 

          </div>`;

      };


      (bases || []).forEach(b => {

        const baseName = (b["基地名稱"] || "").toString();

        const address = (b["地址"] || "").toString();

        const { lat, lng } = parseLatLng(b);

        if (Number.isFinite(lat) && Number.isFinite(lng)) {

          const hasActivity = globalActivities.some(a => normalize(a["活動辦理基地"]) === normalize(baseName));

          const normalIcon = L.divIcon({ html: pinHtml(32, 32, "#65502d", hasActivity), iconSize: [32, 32], iconAnchor: [16, 32], popupAnchor: [0, -32] });

          const selectedIcon = L.divIcon({ html: pinHtml(42, 42, "#e2715d", false), iconSize: [42, 42], iconAnchor: [21, 42], popupAnchor: [0, -42] });

          const m = L.marker([lat, lng], { icon: normalIcon }).addTo(markerGroup);

          m.myNormalIcon = normalIcon;

          const gpsUrl = (b["Google map 連結"] || "").toString() || buildGoogleMapsUrl(lat, lng, address);

          m.bindPopup(`<div class="popup-base-name">${escapeHtml(baseName)}</div><div class="popup-base-addr"><a href="${gpsUrl}" target="_blank" class="gps-link">（Google Map 連結）</a></div>`, { closeButton: false });

            if (lastClickedMarker) lastClickedMarker.setIcon(lastClickedMarker.myNormalIcon);

            m.setIcon(selectedIcon);

            lastClickedMarker = m;

            filterActivitiesByBase(baseName);

          });

        }

      });

      markerGroup.addTo(map);

      if (markerGroup.getLayers().length > 0) map.fitBounds(markerGroup.getBounds(), { padding: [40, 40] });

    }

    function renderCardsByIndex(idxList, targetId, showBaseTag) {

    function openDetailByIdx(idx, opts) {

      opts = opts || {};

      if (!opts.skipPush) history.pushState({ view: "detail", idx: idx, tab: getActiveTab() }, document.title, location.pathname + location.search);

      const a = globalActivities[idx] || {};

      const { lat, lng } = parseLatLng(baseData);

      const gpsUrl = (baseData ? (baseData["Google map 連結"] || "").toString() : "") || buildGoogleMapsUrl(lat, lng, baseData ? baseData["地址"] : "");


      $("det-title").innerText = a["活動名稱"] || "";

      $("det-base-info").innerHTML = `${escapeHtml(baseName)}（<a href="${gpsUrl}" target="_blank" class="gps-link">Google Map 連結</a>）`;

      

      const dt = parseDateForWeekday(a["日期"] || "");

      const fTime = dt ? `${a["日期"]}(${["週日","週一","週二","週三","週四","週五","週六"][dt.getDay()]}) ${a["時間"] || ""}` : `${a["日期"] || ""} ${a["時間"] || ""}`;

      $("det-location").innerText = a["活動地點"] || "基地內空間";

      $("det-content").innerText = a["內容"] || "暫無簡介資訊";

      elOverlay.style.display = "flex";

    }
